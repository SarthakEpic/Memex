    12→// ─────────────────────────────────────────────────────────────────────────────
    13→// Retry wrapper with exponential backoff for 429 (rate limit) errors.
    14→// The z-ai SDK does not retry by default, and bulk operations (decision
    15→// extraction, chat during traffic spikes) can hit 429s. We retry up to 5
    16→// times with 5s, 10s, 20s, 40s, 60s backoff.
    17→// ─────────────────────────────────────────────────────────────────────────────
    18→
    19→const MAX_RETRIES = 5
    20→const BASE_BACKOFF_MS = 5000
    21→
    22→function isRateLimited(err: unknown): boolean {
    23→  const msg = err instanceof Error ? err.message : String(err)
    24→  return msg.includes("429") || msg.toLowerCase().includes("too many requests")
    25→}
    26→
    27→async function withRetry<T>(fn: () => Promise<T>): Promise<{ ok: true; value: T } | { ok: false; error: unknown }> {
    28→  let lastErr: unknown
    29→  for (let attempt = 0; attempt <= MAX_RETRIES; attempt++) {
    30→    try {
    31→      const value = await fn()
    32→      return { ok: true, value }
    33→    } catch (err) {
    34→      lastErr = err
    35→      if (!isRateLimited(err)) return { ok: false, error: err }
    36→      if (attempt === MAX_RETRIES) break
    37→      // Exponential backoff with jitter: 5s, 10s, 20s, 40s, 60s
    38→      const baseBackoff = Math.min(BASE_BACKOFF_MS * Math.pow(2, attempt), 60000)
    39→      const jitter = Math.random() * 2000 // Add randomness to avoid thundering herd
    40→      const backoff = baseBackoff + jitter
    41→      await new Promise((r) => setTimeout(r, backoff))
    42→    }
    43→  }
    44→  return { ok: false, error: lastErr }
    45→}

   224→export async function generateSmartAnswer(
   225→  question: string,
   226→  chunks: ContextChunk[],
   227→  history: { role: "user" | "assistant"; content: string }[] = [],
   228→  emailContext?: string
   229→): Promise<SmartAnswer> {
   230→  // Build a compact context block — only include the most relevant chunks
   231→  // and keep email context SHORT to avoid drowning out conversation history
   232→  const contextBlock = chunks.length > 0 ? buildContextBlock(chunks.slice(0, 3)) : ""
   233→  const shortEmailContext = emailContext ? emailContext.slice(0, 800) : ""
   234→
   235→  // Keep MORE conversation history (last 10 messages = 5 exchanges)
   236→  // This is critical for maintaining conversation flow
   237→  const recentHistory = history.slice(-10)
   238→
   239→  // Build the user prompt — keep it SHORT so conversation history fits
   240→  // The context is provided as SEPARATE messages, not stuffed into the user prompt
   241→  const messages: { role: "system" | "user" | "assistant"; content: string }[] = [
   242→    { role: "system", content: SMART_SYSTEM_PROMPT },
   243→  ]
   244→
   245→  // Add note context as a system-level note (not part of the conversation)
   246→  if (contextBlock) {
   247→    messages.push({
   248→      role: "system",
   249→      content: `NOTE CONTEXT (use ONLY if the user asks about their notes, cite with [^chunk_id]):\n${contextBlock}`,
   250→    })
   251→  }
   252→
   253→  // Add email context as a system-level note (keep it SHORT)
   254→  if (shortEmailContext) {
   255→    messages.push({
   256→      role: "system",
   257→      content: `RECENT INBOX EMAILS (use ONLY if the user asks about emails):\n${shortEmailContext}`,
   258→    })
   259→  }
   260→
   261→  // Add conversation history — this is the KEY part for maintaining flow
   262→  messages.push(...recentHistory.map((h) => ({ role: h.role, content: h.content })))
   263→
   264→  // Add the current user message as a clean, simple message
   265→  // (NOT wrapped in context — the context is already in system messages above)
   266→  messages.push({ role: "user", content: question })
   267→
   268→  const zai = await getClient()
   269→  let raw = ""
   270→  const result = await withRetry(() =>
   271→    zai.chat.completions.create({
   272→      messages,
   273→      temperature: 0.4,
   274→      max_tokens: 800,
   275→    })
   276→  )
   277→  if (!result.ok) {
   278→    return {
   279→      answer:
   280→        "⚠️ I'm having trouble connecting right now (rate limited). Please try again in a moment — your message has been saved.",
   281→      mode: "general",
   282→      citedChunkIds: [],
   283→      refused: false,
   284→      serviceError: true,
   285→    }
   286→  }
   287→  const completion = result.value
   288→  if (typeof completion === "string") {
   289→    raw = completion
   290→  } else if (completion?.choices?.[0]?.message?.content) {
   291→    raw = completion.choices[0].message.content
   292→  } else if (completion?.content) {
   293→    raw = completion.content
   294→  } else {
   295→    raw = String(completion ?? "")
   296→  }
   297→
   298→  // Post-process: extract + verify citations
   299→  const validIds = new Set(chunks.map((c) => c.id))
   300→  const cited = new Set<string>()
   301→  const markerRe = /\[\^([a-z0-9]+)\]/gi
   302→  let m: RegExpExecArray | null
   303→  while ((m = markerRe.exec(raw)) !== null) {
   304→    const id = m[1]
   305→    if (validIds.has(id)) cited.add(id)
   306→  }
   307→
   308→  // Detect mode from the answer
   309→  const lower = raw.toLowerCase()
   310→  let mode: ChatMode = "general"
   311→  if (cited.size > 0) mode = "note_qa"
   312→  else if (lower.includes("i don't have a source") || lower.includes("in your notes")) mode = "note_qa"
   313→  else if (lower.includes("memex can") || lower.includes("you can") || lower.includes("to add a note") || lower.includes("keyboard shortcut") || lower.includes("command palette")) mode = "app_help"
   314→  else if (lower.includes("email") || lower.includes("inbox")) mode = "email"
   315→
   316→  const refused = mode === "note_qa" && cited.size === 0
   317→
   318→  return {
   319→    answer: raw.trim(),
   320→    mode,
   321→    citedChunkIds: Array.from(cited),
   322→    refused,
   323→    serviceError: false,
   324→  }
   325→}

   341→function buildContextBlock(chunks: ContextChunk[]): string {
   342→  return chunks
   343→    .map((c) => {
   344→      return `[chunk_id: ${c.id}]
   345→source: ${c.sourcePath}${c.headingPath ? ` (section: ${c.headingPath})` : ""}
   346→---
   347→${c.text}`
   348→    })
   349→    .join("\n\n")
   350→}

   413→  const markerRe = /\[\^([a-z0-9]+)\]/gi

   476→export async function analyzeEmail(
   477→  from: string,
   478→  subject: string,
   479→  body: string
   480→): Promise<EmailAnalysis | null> {
   481→  const zai = await getClient()
   482→  const result = await withRetry(() =>
   483→    zai.chat.completions.create({
   484→      messages: [
   485→        { role: "system", content: EMAIL_ANALYSIS_PROMPT },
   486→        {
   487→          role: "user",
   488→          content: `FROM: ${from}\nSUBJECT: ${subject}\n\nBODY:\n${body.slice(0, 2000)}\n\nJSON:`,
   489→        },
   490→      ],
   491→      temperature: 0.2,
   492→      max_tokens: 500,
   493→    })
   494→  )
   495→  if (!result.ok) return null
   496→  try {
   497→    const completion = result.value
   498→    let raw = ""
   499→    if (typeof completion === "string") raw = completion
   500→    else if (completion?.choices?.[0]?.message?.content)
   501→      raw = completion.choices[0].message.content
   502→    else raw = String(completion ?? "")
   503→    const match = raw.match(/\{[\s\S]*\}/)
   504→    if (!match) return null
   505→    return JSON.parse(match[0]) as EmailAnalysis
   506→  } catch {
   507→    return null
   508→  }
   509→}

   526→        },
   527→        {
   528→          role: "user",
   529→          content: `ORIGINAL EMAIL FROM: ${from}\nSUBJECT: ${subject}\nBODY: ${body.slice(0, 1500)}\n\nUSER'S INSTRUCTION: ${instruction}\n\nDRAFT REPLY:`,
   530→        },
   531→      ],
   532→      temperature: 0.5,
   533→      max_tokens: 400,
   534→    })
   535→  )
   536→  if (!result.ok) return "Unable to generate a draft at this time."
   537→  const completion = result.value
   538→  let raw = ""
   539→  if (typeof completion === "string") raw = completion
   540→  else if (completion?.choices?.[0]?.message?.content)
   541→    raw = completion.choices[0].message.content
   542→  else raw = String(completion ?? "")
   543→  return raw.trim()
   544→}

   546→// ─────────────────────────────────────────────────────────────────────────────
   547→// LLM reranker — semantic re-ranking of BM25 candidates.
   548→// We ask the model to score 0..1 relevance of each candidate to the query.
   549→// Falls back gracefully if the model misbehaves.
   550→// ─────────────────────────────────────────────────────────────────────────────
   551→
   552→export async function llmRerank(
   553→  query: string,
   554→  chunks: ContextChunk[]
   555→): Promise<ContextChunk[]> {
   556→  if (chunks.length <= 1) return chunks
   557→  const zai = await getClient()
   558→  const numbered = chunks
   559→    .map((c, i) => `[${i}] ${c.text.slice(0, 400)}`)
   560→    .join("\n")
   561→  const prompt = `Rate the relevance of each candidate passage to the query on a scale of 0.0 to 1.0. Respond ONLY with a JSON array of numbers, one per candidate, in order. No prose.
   562→
   563→QUERY: ${query}
   564→
   565→CANDIDATES:
   566→${numbered}
   567→
   568→JSON scores:`
   569→  try {
   570→    const result = await withRetry(() =>
   571→      zai.chat.completions.create({
   572→        messages: [
   573→          {
   574→            role: "system",
   575→            content:
   576→              "You are a relevance rater. Output only a JSON array of floats between 0 and 1.",
   577→          },
   578→          { role: "user", content: prompt },
   579→        ],
   580→        temperature: 0,
   581→        max_tokens: 200,
   582→      })
   583→    )
   584→    if (!result.ok) return chunks
   585→    const completion = result.value
   586→    let raw = ""
   587→    if (typeof completion === "string") raw = completion
   588→    else if (completion?.choices?.[0]?.message?.content)
   589→      raw = completion.choices[0].message.content
   590→    else raw = String(completion ?? "")
   591→    const match = raw.match(/\[[\s\S]*\]/)
   592→    if (!match) return chunks
   593→    const scores = JSON.parse(match[0]) as number[]
   594→    if (!Array.isArray(scores) || scores.length !== chunks.length) return chunks
   595→    return chunks
   596→      .map((c, i) => ({ ...c, score: c.score * 0.4 + (scores[i] ?? 0) * 2 }))
   597→      .sort((a, b) => b.score - a.score)
   598→  } catch {
   599→    return chunks
   600→  }
   601→}

   603→// ─────────────────────────────────────────────────────────────────────────────
   604→// Decision extraction (Phase 8 of the spec)
   605→// ─────────────────────────────────────────────────────────────────────────────
   606→
   607→export interface ExtractedDecision {
   608→  title: string
   609→  decisionDate: string
   610→  alternatives: string[]
   611→  rationale: string
   612→  outcome: string
   613→  participants: string[]
   614→  confidence: number
   615→}
   616→
   617→const DECISION_PROMPT = `You are Memex's decision extractor. Read the following note chunk and extract any decisions it contains. A decision is a statement that X was chosen over Y for reason Z.
   618→
   619→If no decision is present, return {"decisions": []}.
   620→
   621→If decisions are present, return JSON:
   622→{
   623→  "decisions": [
   624→    {
   625→      "title": "Use Postgres for primary DB",
   626→      "decision_date": "2024-03-14",
   627→      "alternatives": ["Mongo", "SQLite"],
   628→      "rationale": "ACID guarantees matter for financial data",
   629→      "outcome": "Working well 6 months later",
   630→      "participants": ["@me"],
   631→      "confidence": 0.9
   632→    }
   633→  ]
   634→}
   635→
   636→Respond with JSON only, no prose. Dates must be ISO format YYYY-MM-DD or null.`
   637→
   638→export async function extractDecisions(
   639→  chunkText: string,
   640→  headingPath: string
   641→): Promise<ExtractedDecision[]> {
   642→  const zai = await getClient()
   643→  const result = await withRetry(() =>
   644→    zai.chat.completions.create({
   645→      messages: [
   646→        { role: "system", content: DECISION_PROMPT },
   647→        {
   648→          role: "user",
   649→          content: `Heading path: ${headingPath}\n\nNOTE CHUNK:\n${chunkText}\n\nJSON:`,
   650→        },
   651→      ],
   652→      temperature: 0.1,
   653→      max_tokens: 600,
   654→    })
   655→  )
   656→  if (!result.ok) return []
   657→  try {
   658→    const completion = result.value
   659→    let raw = ""
   660→    if (typeof completion === "string") raw = completion
   661→    else if (completion?.choices?.[0]?.message?.content)
   662→      raw = completion.choices[0].message.content
   663→    else raw = String(completion ?? "")
   664→    const match = raw.match(/\{[\s\S]*\}/)
   665→    if (!match) return []
   666→    const parsed = JSON.parse(match[0]) as { decisions?: ExtractedDecision[] }
   667→    if (!parsed.decisions || !Array.isArray(parsed.decisions)) return []
   668→    return parsed.decisions.filter((d) => d && d.title && d.rationale)
   669→  } catch {
   670→    return []
   671→  }
   672→}
