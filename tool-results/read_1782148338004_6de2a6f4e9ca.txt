   192тЖТ   192тЖТ   192тЖТ### Database Schema (Alembic migration)
   193тЖТ   193тЖТ   193тЖТ
   194тЖТ   194тЖТ   194тЖТ```sql
   195тЖТ   195тЖТ   195тЖТ-- chunks: the atomic unit of retrieval
   196тЖТ   196тЖТ   196тЖТCREATE TABLE chunks (
   197тЖТ   197тЖТ   197тЖТ    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
   198тЖТ   198тЖТ   198тЖТ    doc_id UUID NOT NULL,                -- groups chunks from same file

   200тЖТ   200тЖТ   200тЖТ    source_type TEXT NOT NULL,           -- 'markdown' | 'pdf' | 'commit'
   201тЖТ   201тЖТ   201тЖТ    chunk_index INT NOT NULL,            -- 0,1,2,... within doc
   202тЖТ   202тЖТ   202тЖТ    text TEXT NOT NULL,
   203тЖТ   203тЖТ   203тЖТ    tsv TSVECTOR GENERATED ALWAYS AS (to_tsvector('english', text)) STORED,

   347тЖТ   347тЖТ   347тЖТdef run_eval():
   348тЖТ   348тЖТ   348тЖТ    golden = [json.loads(l) for l in open('eval/golden_set.jsonl')]
   349тЖТ   349тЖТ   349тЖТ    results = []

   354тЖТ   354тЖТ   354тЖТ        recall = len(set(retrieved_ids) & set(item['golden_chunk_ids'])) / len(item['golden_chunk_ids'])

   360тЖТ   360тЖТ   360тЖТ                break

   363тЖТ   363тЖТ   363тЖТ    # Aggregate
   364тЖТ   364тЖТ   364тЖТ    avg_recall = sum(r['recall'] for r in results) / len(results)
   365тЖТ   365тЖТ   365тЖТ    avg_mrr = sum(r['mrr'] for r in results) / len(results)
   366тЖТ   366тЖТ   366тЖТ    p95_latency = percentile([r['latency_ms'] for r in results], 95)
   367тЖТ   367тЖТ   367тЖТ
   368тЖТ   368тЖТ   368тЖТ    print(f"Recall@5: {avg_recall:.3f}")
   369тЖТ   369тЖТ   369тЖТ    print(f"MRR:      {avg_mrr:.3f}")

   371тЖТ   371тЖТ   371тЖТ

   400тЖТ   400тЖТ   400тЖТ### Implementation
   401тЖТ   401тЖТ   401тЖТ
   402тЖТ   402тЖТ   402тЖТ1. **`services/retrieval.py`** тАФ replace naive version with hybrid:
   403тЖТ   403тЖТ   403тЖТ
   404тЖТ   404тЖТ   404тЖТ   ```python
   405тЖТ   405тЖТ   405тЖТ   async def hybrid_search(query: str, top_k: int = 20) -> list[Chunk]:
   406тЖТ   406тЖТ   406тЖТ       # 1. BM25 (Postgres FTS)
   407тЖТ   407тЖТ   407тЖТ       bm25_results = await db.execute("""
   408тЖТ   408тЖТ   408тЖТ           SELECT id, ts_rank(tsv, plainto_tsquery('english', :q)) AS score
   409тЖТ   409тЖТ   409тЖТ           FROM chunks
   410тЖТ   410тЖТ   410тЖТ           WHERE tsv @@ plainto_tsquery('english', :q)
   411тЖТ   411тЖТ   411тЖТ           ORDER BY score DESC
   412тЖТ   412тЖТ   412тЖТ           LIMIT 50
   413тЖТ   413тЖТ   413тЖТ       """, {"q": query})
   414тЖТ   414тЖТ   414тЖТ       bm25_ranked = [r.id for r in bm25_results]  # ordered list
   415тЖТ   415тЖТ   415тЖТ
   416тЖТ   416тЖТ   416тЖТ       # 2. Vector (Qdrant)
   417тЖТ   417тЖТ   417тЖТ       query_emb = await embeddings_client.embed([query])
   418тЖТ   418тЖТ   418тЖТ       vec_results = await qdrant.search(
   419тЖТ   419тЖТ   419тЖТ           collection="memex_chunks",
   420тЖТ   420тЖТ   420тЖТ           query_vector=query_emb[0],
   421тЖТ   421тЖТ   421тЖТ           limit=50
   422тЖТ   422тЖТ   422тЖТ       )
   423тЖТ   423тЖТ   423тЖТ       vec_ranked = [r.id for r in vec_results]
   424тЖТ   424тЖТ   424тЖТ
   425тЖТ   425тЖТ   425тЖТ       # 3. Reciprocal Rank Fusion (k=60)
   426тЖТ   426тЖТ   426тЖТ       K = 60
   427тЖТ   427тЖТ   427тЖТ       fused_scores = {}
   428тЖТ   428тЖТ   428тЖТ       for rank, cid in enumerate(bm25_ranked, 1):
   429тЖТ   429тЖТ   429тЖТ           fused_scores[cid] = fused_scores.get(cid, 0) + 1 / (K + rank)
   430тЖТ   430тЖТ   430тЖТ       for rank, cid in enumerate(vec_ranked, 1):
   431тЖТ   431тЖТ   431тЖТ           fused_scores[cid] = fused_scores.get(cid, 0) + 1 / (K + rank)
   432тЖТ   432тЖТ   432тЖТ
   433тЖТ   433тЖТ   433тЖТ       # 4. Sort by fused score, take top 20
   434тЖТ   434тЖТ   434тЖТ       top_ids = sorted(fused_scores, key=fused_scores.get, reverse=True)[:top_k]
   435тЖТ   435тЖТ   435тЖТ       return await chunk_repo.get_many(top_ids)
   436тЖТ   436тЖТ   436тЖТ   ```
